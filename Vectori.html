Un vector arata cam asa:
Acesta este un vector cu numarul maxim de spatii in memorie egal cu 10. In prima faza, aceste spatii sunt goale. Trebuie sa le atribuim numere.
In general vectorii se noteaza cu litere mici(ex: v, a, b, etc).


Declararea unui vector:
Daca avem un vector cu 100 de elemente, initializam vectorul cu 1 mai mult pentru a incepe numerotarea pozitiilor de la 1 nu de la 0.
int v[101];









Ca sa ne fie mai usor sa ne imaginam un vector, spunem ca un vector definit cu 101, reprezinta un dulap cu 100 de sertare.
Fiecare sertar avand alocat pentru un singur numar.

Pentru a lucra cu un vector trebuie sa sa-l citim si daca este nevoie si sa-l afisam. 

                int main()
                {
                    int v[101],n,i;

                    cin>>n; // n reprezinta numarul de elemente din vector
                    for(i=1; i<=n; i++)
                        cin>>v[i];

                    // Prelucram vectorul

                    for(i=1; i<=n; i++)
                        cout<< v[i]<<" ";
                }

Sortarea unui vector, este un algoritm de baza ce trebuie stiut foarte bine!
Sortarea poate fi crescatoare sau descrescatoare, singura diferenta este de conditia pusa in cele 2 for-uri. 

                int main()
                {
                    int v[101],n,i,aux;
                    cin>>n;
                    for(i=1; i<=n; i++)
                        cin>>v[i];

                    for(i=1; i< n; i++)
                        for(j=i+1; j<=n; j++)
                            if(v[i] > v[j]) // Asta e conditia pentru sortare crescatoare, pentru cea descrescatoare ( if(v[i] < v[j] ) )
                            {
                                aux=v[i];
                                v[i]=v[j]; // Putem inlocui toate astea cu  swap(v[i],v[j]) 
                                v[j]=aux;
                            }
                }

Minimul si maximul unui vector se realizeaza la fel ca la un sir de nuere, la vectori putem memora si pozitia minimului/maximului. 

                int main()
                {
                    int v[101],n,i,p,q;
                    int maxim = 0; //Initializam maximul cu cea mai mica valoare posibila ;
                    int minim = 999999; //Initializam minimul cu cea mai mare valoare posibila;

                    cin>>n;
                    for(i=1; i<=n; i++)
                        cin>>v[i];

                    for(i=1; i<=n; i++)
                        {
                            if(maxim < v[i])
                                {
                                    maxim = v[i];
                                    p=i; //P-ul va retine pozitia maximului din vector;
                                }
                            if(minim > v[i])
                                {
                                    minim = v[i];
                                    q=i; //Q-ul va retine pozitia minimului din vector;
                                }
                        }
                    cout<<"Maximul este "<< maxim<<" si este situat pe pozitia "<< p;
                    cout<<"Minimul este "<< minim<<" si este situat pe pozitia "<< q;

                    return 0;
                }

Pentru a insercta doi vectori ne vom folosi de urmatorul algoritm care verifica daca doua elemente sunt egale. 

               int main()
               {
                int a[101], b[101], n, m, i, j, c[101], k=0;

                cin>>n;
                for(i=1; i<=n; i++)
                    cin>>a[i];
                cin>>m;
                for(j=1; j<=m; j++)
                    cin>>b[j];
                for(i=1; i<=n; i++)
                    for(j=1; j<=m; j++)
                        if(a[i] == b[j]) //Daca sunt 2 elemente egale le punem intr-un alt vector
                            c[k++]=a[i];
                for(i=1; i<=k; i++)
                    cout<< c[i]<<" ";
               }

Ca sa reunim 2 vectori, facem un al 3-lea vector si copiem cei 2 vectori in el.

                int main()

Algoritmul de permutare circulara cate o pozitie spre stanga este pur si simplu o rotatie a vectorului.
E ca si cum am inversa indicii(in loc sa fie de la 1 la 10 poate sa fie de la 5 la 4) 

               
                int main()
                {
                    int v[101],n,i,aux;
                    cin>>n;
                    for(i=1;i<=n;i++)
                        cin>>v[i];
                    for(i=1;i<=n-1;i++)
                    {
                        aux=v[i];
                        v[i]=v[i+1];
                        v[i+1]=aux;
                    }
                    for(i=1;i<=n;i++)
                        cout<< v[i]<<" ";
                    return 0;
                }

Algoritmul de permutare circulara cate o pozitie spre dreapta este la fel cu cel spre stanga doar difera sensul. 

                   
                int main()
                {
                    int v[101],n,i,aux;
                    cin>>n;
                    for(i=1;i<=n;i++)
                        cin>>v[i];
                    for(i=n;i>=2;i--)
                    {
                        aux=v[i];
                        v[i]=v[i-1];
                        v[i-1]=aux;
                    }
                    for(i=1;i<=n;i++)
                        cout<< v[i]<<" ";
                    return 0;
                }
